print("Enter numbers. Type -1 to stop.")

prime_count = 0

while True:
    number = int(input())
    is_prime = 1

    if number > 1:
        divisor = 2
        while divisor < number:
            if number % divisor == 0:
                is_prime = 0
            divisor += 1

        if is_prime == 1:
            prime_count += 1

    if number == -1 :
        break


if prime_count % 2:
    print("Odd")
else:
    print("Even")
